本页面将引用jQuery，然后操作DOM。

注意使用jQuery时可以自己传入参数从而使用局部变量，也可以省略，从而使用全局的$。这是因为jQuery自己注册了全局变量$和jQuery的缘故，更多信息点击此处。


        requirejs(['../common'], function (common) {

            // 依赖数组中引入jquery
            requirejs(['jquery'], function ($) {
                $(function () {
                    $("#testdiv").text("我是自动生成的！");
                });
            });

        });
